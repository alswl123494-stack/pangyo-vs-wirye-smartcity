# -*- coding: utf-8 -*- """ 토지이용(용도지역) 구성비 — 반경 클리핑 버전 (버그 수정) 이전 스크립트는 반경에 "걸치는" 필지를 통째로 계산해서 면적이 부풀려졌음. 이번엔 분석원(반경 1.2km)으로 실제로 잘라낸(intersection) 면적만 계산함. 실행 전: pip install requests shapely """ import math import requests from shapely.geometry import Point, Polygon, MultiPolygon from shapely.ops import unary_union from collections import defaultdict VWORLD_KEY = "여기에_VWorld_인증키" DOMAIN = "alswl123494-stack.github.io" RADIUS_M = 1200 REGIONS = { "pangyo": (127.1112, 37.3948), "cheongna": (126.6246, 37.5565), } VWORLD_URL = "https://api.vworld.kr/req/data" HEADERS = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 " "(KHTML, like Gecko) Chrome/124.0.0.0 Safari/537.36", } def project(coords, lng0, lat0): mx = 111320 * math.cos(math.radians(lat0)) my = 110540 return [((x - lng0) * mx, (y - lat0) * my) for x, y in coords] def to_shapely(geometry, lng0, lat0): if geometry["type"] == "Polygon": rings = geometry["coordinates"] shell = project(rings[0], lng0, lat0) holes = [project(r, lng0, lat0) for r in rings[1:]] return Polygon(shell, holes) elif geometry["type"] == "MultiPolygon": polys = [] for rings in geometry["coordinates"]: shell = project(rings[0], lng0, lat0) holes = [project(r, lng0, lat0) for r in rings[1:]] polys.append(Polygon(shell, holes)) return MultiPolygon(polys) return None def vworld_get_features(data_code, lng, lat, radius): features = [] page = 1 size = 1000 while True: params = { "service": "data", "request": "GetFeature", "data": data_code, "key": VWORLD_KEY, "domain": DOMAIN, "geomFilter": f"POINT({lng} {lat})", "buffer": radius, "geometry": "true", "attribute": "true", "crs": "EPSG:4326", "format": "json", "size": size, "page": page, } res = requests.get(VWORLD_URL, params=params, headers=HEADERS, timeout=30).json() result = res.get("response", {}).get("result") if not result: print(" 응답 이상:", res.get("response")) break feats = result.get("featureCollection", {}).get("features", []) features.extend(feats) total = int(res["response"]["record"]["total"]) if page * size >= total or not feats: break page += 1 return features def main(): rows = [] for region, (lng, lat) in REGIONS.items(): print(f"[{region}] 용도지역지구(LT_C_UQ111) 가져오는 중...") feats = vworld_get_features("LT_C_UQ111", lng, lat, RADIUS_M) print(f" {len(feats)}개 필지 (클리핑 전)") circle = Point(0, 0).buffer(RADIUS_M, resolution=64) circle_area = circle.area print(f" 분석원 면적: {circle_area:,.0f} m^2 (반경 {RADIUS_M}m)") by_type = defaultdict(float) for f in feats: uname = f["properties"].get("uname", "미분류") geom = to_shapely(f["geometry"], lng, lat) if geom is None or not geom.is_valid: geom = geom.buffer(0) if geom else None if geom is None: continue clipped = geom.intersection(circle) by_type[uname] += clipped.area total_clipped = sum(by_type.values()) print(f" 클리핑 후 합계: {total_clipped:,.0f} m^2 (분석원의 {total_clipped/circle_area*100:.1f}%)") for uname, area in sorted(by_type.items(), key=lambda x: -x[1]): pct_of_circle = area / circle_area * 100 rows.append({"region": region, "uname": uname, "area_m2": area, "pct_of_circle": pct_of_circle}) import csv with open("out_landuse_summary_fixed.csv", "w", newline="", encoding="utf-8-sig") as fp: w = csv.DictWriter(fp, fieldnames=["region", "uname", "area_m2", "pct_of_circle"]) w.writeheader() for r in rows: w.writerow({"region": r["region"], "uname": r["uname"], "area_m2": round(r["area_m2"], 1), "pct_of_circle": round(r["pct_of_circle"], 2)}) print("\n저장 완료: out_landuse_summary_fixed.csv") if __name__ == "__main__": main()
